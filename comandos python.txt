// crear un entorno virtual
python -m venv env

// activar entorno virtual
env\Scripts\activate

// desactivar el entorno virtual
deactivate

// importante para que el código se vea bien
pip install autopep8

// que dependencias y versiones tiene el entrono virtual
pip freeze

// crear archivo de requerimientos
pip freeze > "requirements.txt"

// recrear un entorno (instalar requerimientos) 
pip install -r "requirements.txt"

Django

// crear un proyecto en django
django-admin startproject mysite

// para migrar toda la información a la DB
Py manage.py migrate

// acceso al servidor de desarrollo
py manage.py runserver

// crear una aplicacion dentro de un proyecto
py manage.py startapp myapp

// chequear si una app esta intalada
py manage.py check myapp

// para solicitar a python que haga las migraciones luedo de hacer las clases en modelos
py manage.py makemigrations

// para crear el super usuario y poder entrar el en panel de administración de nuestra app
py manage.py createsuperuser

crear un ejecutable

para instalar pyinstaller en nuestro env
pip install pyinstaller

para ejecutar y crear el ejecutable si solo tenemos un script
pyinstaller your_program.py

para varios scripts
pyi-makespec main.py --windowed

al crear el achivo
en data -> ('./img/*.ico', 'img'), ('./database/*.db', 'database')

para crear el ejecutable 
pyinstaller main.spec
